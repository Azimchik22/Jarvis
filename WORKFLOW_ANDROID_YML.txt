name: Build Android APK
on:
  workflow_dispatch:
  push:
    branches: [ "main" ]
jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Build with Buildozer
        uses: ArtemSBulgakov/buildozer-action@v1
        with:
          work-dir: .
          buildozer-cmd: buildozer android debug
      - name: Upload APK
        uses: actions/upload-artifact@v4
        with:
          name: JarvisAI-APK
          path: bin/*.apk
